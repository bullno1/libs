915bent_bitset_clear(bent_bitset_t* bitset) {

919static void
920bent_bitset_set(bent_bitset_t* bitset, bent_index_t bit_index) {
921 bent_index_t num_bits_per_mask = sizeof(bent_index_t) * CHAR_BIT;
922 bent_index_t mask_index = bit_index / num_bits_per_mask;
923 bent_mask_t mask = (bent_mask_t)1 << (bit_index % num_bits_per_mask);
924 bitset->bits[mask_index] |= mask;
925}
926
927static void
928bent_bitset_unset(bent_bitset_t* bitset, bent_index_t bit_index) {
929 bent_index_t num_bits_per_mask = sizeof(bent_index_t) * CHAR_BIT;
930 bent_index_t mask_index = bit_index / num_bits_per_mask;
931 bent_mask_t mask = ~((bent_mask_t)1 << (bit_index % num_bits_per_mask));
932 bitset->bits[mask_index] &= mask;
933}

935static void
936bent_bitset_flip(bent_bitset_t* bitset) {
937 for (bent_index_t i = 0; i < BENT_BITSET_LEN; ++i) {
938 bitset->bits[i] = ~bitset->bits[i];
939 }
940}

942static bool
943bent_bitset_check(const bent_bitset_t* bitset, bent_index_t bit_index) {
944 bent_index_t num_bits_per_mask = sizeof(bent_index_t) * CHAR_BIT;
945 bent_index_t mask_index = bit_index / num_bits_per_mask;
946 bent_mask_t mask = (bent_mask_t)1 << (bit_index % num_bits_per_mask);
947 return (bitset->bits[mask_index] & mask) > 0;
948}
949
950static bool
951bent_bitset_any_match(const bent_bitset_t* subject, const bent_bitset_t* requirement) {
952 bool result = false;
953 for (bent_index_t i = 0; i < BENT_BITSET_LEN; ++i) {
954 bent_mask_t subject_mask = subject->bits[i];
955 bent_mask_t required_mask = requirement->bits[i];
956 result = result || ((subject_mask & required_mask) > 0);
957 }
958 return result;
959}
960
961static bool
962bent_bitset_all_match(const bent_bitset_t* subject, const bent_bitset_t* requirement) {
963 bool result = true;
964 for (bent_index_t i = 0; i < BENT_BITSET_LEN; ++i) {
965 bent_mask_t subject_mask = subject->bits[i];
966 bent_mask_t required_mask = requirement->bits[i];
967 result = result && ((subject_mask & required_mask) == required_mask);
968 }
969 return result;
970}

1036static void
1037bent_sys_init(
1038 bent_world_t* world,
1039 bent_system_data_t* sys,
1040 const char* name, const bent_sys_def_t* def
1041) {
1042 sys->def = def;
1043
1044 BENT_LOG("Initializing %s", name);
1045
1046#ifndef BENT_NO_RELOAD
1047 // Backup old filter to detect change
1048 bent_bitset_t old_require = sys->require;
1049 bent_bitset_t old_exclude = sys->exclude;
1050#endif
1051
1052 if (sys->name == NULL) {
1053 sys->name = bent_strcpy(name, world->memctx);
1054 }
1055
1056 bent_bitset_clear(&sys->require);
1057 bent_bitset_clear(&sys->exclude);
1058
1059 if (def->require == NULL && def->exclude == NULL) {
1060 // A system that specifies nothing will not match anything
1061 bent_bitset_flip(&sys->require);
1062 bent_bitset_flip(&sys->exclude);
1063 } else {
1064 // Otherwise, each property is defaulted to an empty list
1065 for (
1066 bent_comp_reg_t** comp = def->require;
1067 comp != NULL && *comp != NULL;
1068 ++comp
1069 ) {
1070 bent_bitset_set(&sys->require, (*comp)->id - 1);
1071 }
1072
1073 for (
1074 bent_comp_reg_t** comp = def->exclude;
1075 comp != NULL && *comp != NULL;
1076 ++comp
1077 ) {
1078 bent_bitset_set(&sys->exclude, (*comp)->id - 1);
1079 }
1080 }
1081
1082 bool initialized = sys->initialized;
1083 if (def->size) {
1084 sys->userdata = BENT_REALLOC(sys->userdata, def->size, world->memctx);
1085 if (!initialized) {
1086 memset(sys->userdata, 0, def->size);
1087 }
1088 }
1089
1090 if (def->init && (!initialized || def->allow_reinit)) {
1091 def->init(sys->userdata, world);
1092 }
1093
1094#ifndef BENT_NO_RELOAD
1095 // Update component inclusion based on new filter
1096 // Also, allow newly created systems to immediately register existing entities
1097 bent_system_data_t old_sys = {
1098 .require = old_require,
1099 .exclude = old_exclude,
1100 };
1101 bent_index_t num_entities = (bent_index_t)barray_len(world->entities);
1102 for (bent_index_t i = 0; i < num_entities; ++i) {
1103 const bent_entity_data_t* entity = &world->entities[i];
1104 if (world->entities[i].destroyed) { continue; }
1105
1106 const bent_bitset_t* components = &entity->components;
1107 if (sys->initialized) { // Existing system, do diff
1108 if (bent_sys_match_impl(&old_sys, components)) {
1109 if (!bent_sys_match_impl(sys, components)) {
1110 bent_sys_remove_entity(world, sys, (bent_t){
1111 .index = i + 1,
1112 .gen = entity->generation,
1113 });
1114 }
1115 } else {
1116 if (bent_sys_match_impl(sys, components)) {
1117 bent_sys_add_entity(world, sys, (bent_t){
1118 .index = i + 1,
1119 .gen = entity->generation,
1120 });
1121 }
1122 }
1123 } else { // Newly registered system, try to match existing entities
1124 if (bent_sys_match_impl(sys, components)) {
1125 bent_sys_add_entity(world, sys, (bent_t){
1126 .index = i + 1,
1127 .gen = entity->generation,
1128 });
1129 }
1130 }
1131 }
1132#endif
1133
1134 sys->initialized = true;
1135}

1285 }
