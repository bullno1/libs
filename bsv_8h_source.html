414static inline bsv_status_t
415bsv_read(bsv_in_t* in, void* buf, size_t size) {
416 char* cbuf = buf;
417 while (size > 0) {
418 size_t bytes_read = in->read(in, cbuf, size);
419 if (bytes_read == 0) { return BSV_IO_ERROR; }
420 cbuf += bytes_read;
421 size -= bytes_read;
422 }
423
424 return BSV_OK;
425}
426
427static inline bsv_status_t
428bsv_write(bsv_out_t* out, const void* buf, size_t size) {
429 const char* cbuf = buf;
430 while (size > 0) {
431 size_t bytes_written = out->write(out, cbuf, size);
432 if (bytes_written == 0) { return BSV_IO_ERROR; }
433 cbuf += bytes_written;
434 size -= bytes_written;
435 }
436
437 return BSV_OK;
438}

706bsv_status_t
707bsv_write_uint(uint64_t x, bsv_out_t* out) {
708 char buf[10];
709 size_t n = 0;
710
711 for (int i = 0; i < 10; ++i) {
712 n += x >= 0x80;
713 buf[i] = (char)(x | 0x80);
714 x >>= 7;
715 }
716
717 buf[n] ^= 0x80;
718 n += 1;
719
720 return bsv_write(out, buf, n);
721}

723bsv_status_t
724bsv_write_sint(int64_t x, bsv_out_t* out) {
725 uint64_t ux = ((uint64_t)x << 1) ^ (uint64_t)(x >> 63); // zigzag encode
726 return bsv_write_uint(ux, out);
727}

729bsv_status_t
730bsv_read_uint(uint64_t* x, bsv_in_t* in) {
731 uint64_t b;
732 char c;
733 uint64_t tmp = 0;
734
735 for (int i = 0; i < 10; ++i) {
736 BSV_CHECK_STATUS(bsv_read(in, &c, 1));
737
738 b = c;
739 tmp |= (b & 0x7f) << (7 * i);
740 if (b < 0x80) {
741 *x = tmp;
742 return BSV_OK;
743 }
744 }
745
746 return BSV_MALFORMED;
747}

750bsv_read_sint(int64_t* x, bsv_in_t* in) {
751 uint64_t ux;
752 BSV_CHECK_STATUS(bsv_read_uint(&ux, in));
753 *x = (int64_t)((ux >> 1) ^ (~(ux & 1) + 1)); // zigzag decode
