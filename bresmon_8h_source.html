577int
578bresmon_should_reload(bresmon_t* mon, bool wait) {
579 int num_events = 0;
580
581#if defined(__linux__)
582 if (wait) {
583 struct pollfd pollfd = {
584 .fd = mon->inotifyfd,
585 .events = POLLIN,
586 };
587 poll(&pollfd, 1, -1);
588 }
589
590 _Alignas(struct inotify_event) char event_buf[sizeof(struct inotify_event) + NAME_MAX + 1];
591
592 while (true) {
593 ssize_t num_bytes_read = read(mon->inotifyfd, event_buf, sizeof(event_buf));
594
595 if (num_bytes_read <= 0) {
596 break;
597 }
598
599 for (
600 char* event_itr = event_buf;
601 event_itr - event_buf < num_bytes_read;
602 ) {
603 struct inotify_event* event = (struct inotify_event*)event_itr;
604 event_itr += sizeof(struct inotify_event) + event->len;
605
606 for (
607 bresmon_dirmon_link_t* itr = mon->dirmons.next;
608 itr != &mon->dirmons;
609 itr = itr->next
610 ) {
611 bresmon_dirmon_t* dirmon = (bresmon_dirmon_t*)((char*)itr - offsetof(bresmon_dirmon_t, link));
612
613 if (dirmon->watchd == event->wd) {
614 for (
615 bresmon_watch_link_t* watch_itr = dirmon->watches.next;
616 watch_itr != &dirmon->watches;
617 watch_itr = watch_itr->next
618 ) {
619 bresmon_watch_t* watch = (bresmon_watch_t*)((char*)watch_itr - offsetof(bresmon_watch_t, link));
620 if (strcmp(watch->filename, event->name) == 0) {
621 ++watch->latest_version;
622 ++num_events;
623 }
624 }
625
626 break;
627 }
628 }
629 }
630 }
631#elif defined(_WIN32)
632 OVERLAPPED_ENTRY overlapped_entry;
633 ULONG num_entries = 0;
634
635 while (true) {
636 BOOL dequeued = GetQueuedCompletionStatusEx(
637 mon->iocp,
638 &overlapped_entry, 1,
639 &num_entries,
640 wait ? INFINITE : 0,
641 TRUE
642 );
643 wait = false; // Only wait for the initial event
644
645 if (!dequeued || num_entries == 0) {
646 // No event
647 break;
648 } else if (
649 overlapped_entry.lpOverlapped == NULL
650 || overlapped_entry.dwNumberOfBytesTransferred == 0
651 ) {
652 // Failed
653 continue;
654 }
655
656 bresmon_dirmon_t* dirmon = (bresmon_dirmon_t *)overlapped_entry.lpCompletionKey;
657
658 for (
659 FILE_NOTIFY_INFORMATION* notification_itr = (FILE_NOTIFY_INFORMATION*)dirmon->notification_buf;
660 notification_itr != NULL;
661 notification_itr = notification_itr->NextEntryOffset != 0
662 ? (FILE_NOTIFY_INFORMATION*)((char*)notification_itr + notification_itr->NextEntryOffset)
663 : NULL
664 ) {
665 if (notification_itr->Action == FILE_ACTION_RENAMED_OLD_NAME) { continue; }
666
667 for (
668 bresmon_watch_link_t* watch_itr = dirmon->watches.next;
669 watch_itr != &dirmon->watches;
670 watch_itr = watch_itr->next
671 ) {
672 bresmon_watch_t* watch = (bresmon_watch_t*)((char*)watch_itr - offsetof(bresmon_watch_t, link));
673 if (
674 watch->filename_len == (notification_itr->FileNameLength / sizeof(wchar_t))
675 && wcsncmp(watch->filename, notification_itr->FileName, watch->filename_len) == 0
676 ) {
677 ++watch->latest_version;
678 ++num_events;
679 }
680 }
681 }
682
683 // Queue another read
684 dirmon->overlapped = (OVERLAPPED){ 0 };
685 ReadDirectoryChangesW(
686 dirmon->dir_handle,
687 dirmon->notification_buf,
688 sizeof(dirmon->notification_buf),
689 FALSE,
690 FILE_NOTIFY_CHANGE_FILE_NAME
691 | FILE_NOTIFY_CHANGE_LAST_WRITE,
692 NULL,
693 &dirmon->overlapped,
694 NULL
695 );
696 }
697#endif
698
699 return num_events;
700}
