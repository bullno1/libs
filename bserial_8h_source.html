154bserial_read(bserial_in_t* in, void* buf, size_t size) {
155 char* cbuf = buf;
156 while (size > 0) {
157 size_t bytes_read = in->read(in, cbuf, size);
158 if (bytes_read == 0) { return BSERIAL_IO_ERROR; }
159 cbuf += bytes_read;
160 size -= bytes_read;
161 }
162
163 return BSERIAL_OK;
164}

166static inline bserial_status_t
167bserial_skip(bserial_in_t* in, size_t size) {
168 if (in->skip) {
169 return in->skip(in, size) ? BSERIAL_OK : BSERIAL_IO_ERROR;
170 } else {
171 char buf[BSERIAL_SKIP_BLKSIZE];
172 while (size > 0) {
173 size_t read_size = BSERIAL_SKIP_BLKSIZE > size ? BSERIAL_SKIP_BLKSIZE : size;
174 BSERIAL_CHECK_STATUS(bserial_read(in, buf, read_size));
175 size -= read_size;
176 }
177 return BSERIAL_OK;
178 }

555bserial_status_t
556bserial_write_uint(uint64_t x, bserial_out_t* out) {
557 char buf[10];
558 size_t n = 0;
559
560 for (int i = 0; i < 10; ++i) {
561 n += x >= 0x80;
562 buf[i] = (char)(x | 0x80);
563 x >>= 7;
564 }
565
566 buf[n] ^= 0x80;
567 n += 1;
568
569 return bserial_write(out, buf, n);
570}

580bserial_read_uint(uint64_t* x, bserial_in_t* in) {
581 uint64_t b;
582 char c;

772static inline size_t
773bserial_ctx_mem_layout(void* mem, bserial_ctx_config_t config) {
774 mem_layout_t layout = { 0 };
775 mem_layout_reserve(&layout, sizeof(bserial_ctx_t), _Alignof(bserial_ctx_t));
776
777 ptrdiff_t symtab = mem_layout_reserve(
778 &layout,
779 sizeof(bserial_symbol_t) * config.max_num_symbols,
780 _Alignof(bserial_symbol_t)
781 );
782
783 int32_t symtab_exp = 2;
784 while (((int32_t)1 << symtab_exp) < (int32_t)(config.max_num_symbols * 2)) {
785 ++symtab_exp;
786 }
787 int32_t symtab_index_len = ((int32_t)1 << symtab_exp);
788 ptrdiff_t symtab_index = mem_layout_reserve(
789 &layout,
790 sizeof(int32_t) * symtab_index_len,
791 _Alignof(int32_t)
792 );
793 ptrdiff_t schema_pool = mem_layout_reserve(
794 &layout,
795 sizeof(bserial_record_mapping_t) * config.max_depth * config.max_record_fields,
796 _Alignof(bserial_record_mapping_t)
797 );
798
799 ptrdiff_t scope = mem_layout_reserve(
800 &layout,
801 sizeof(bserial_scope_t) * config.max_depth,
802 _Alignof(bserial_scope_t)
803 );
804
805 ptrdiff_t strpool = mem_layout_reserve(
806 &layout,
807 sizeof(char) * (config.max_symbol_len + 1) * config.max_num_symbols,
808 _Alignof(char)
809 );
810
811 if (mem) {
812 bserial_ctx_t* ctx = mem;
813 ctx->symtab = mem_layout_locate(mem, symtab);
814
815 ctx->symtab_index = mem_layout_locate(mem, symtab_index);
816 ctx->symtab_exp = symtab_exp;
817 memset(ctx->symtab_index, 0, sizeof(*ctx->symtab_index) * symtab_index_len);
818
819 ctx->scope_first = ctx->scope = mem_layout_locate(mem, scope);
820 ctx->scope_last = ctx->scope + config.max_depth - 1;
821 ctx->scope->type = BSERIAL_SCOPE_ROOT;
822 ctx->scope->prev_schema_pool = ctx->schema_pool;
823
824 ctx->strpool = mem_layout_locate(mem, strpool);
825 ctx->schema_pool = mem_layout_locate(mem, schema_pool);
826 ctx->marker_buf = UINT8_MAX;
827 }
828
829 return mem_layout_size(&layout);
830}

872static inline bserial_status_t
873bserial_push_scope(bserial_ctx_t* ctx, bserial_scope_type_t type) {
874 BSERIAL_CHECK_STATUS(ctx->status);
875 if (ctx->scope == ctx->scope_last) {
876 return bserial_malformed(ctx);
877 }
878
879 *(++ctx->scope) = (bserial_scope_t){
880 .type = type,
881 .prev_schema_pool = ctx->schema_pool,
882 };
883
884 if (type == BSERIAL_SCOPE_RECORD || type == BSERIAL_SCOPE_TABLE) {
885 ctx->scope->record_schema = ctx->schema_pool;
886 ctx->schema_pool += ctx->config.max_record_fields;
887 }
888
889 return BSERIAL_OK;
890}

904static inline bserial_status_t
905bserial_begin_op(bserial_ctx_t* ctx, bserial_op_type_t op) {
906 BSERIAL_CHECK_STATUS(ctx->status);
907
908 bserial_scope_t* scope = ctx->scope;
909 bserial_scope_type_t scope_type = scope->type;
910
911 // Can't do anything before filling the blob
912 if (scope_type == BSERIAL_SCOPE_BLOB) {
913 return bserial_malformed(ctx);
914 }
915
916 if (
917 scope_type == BSERIAL_SCOPE_TABLE
918 && op != BSERIAL_OP_RECORD
919 ) {
920 return bserial_malformed(ctx);
921 }
922
923 // Count the number of elements
924 if (
925 scope_type == BSERIAL_SCOPE_ARRAY
926 || scope_type == BSERIAL_SCOPE_TABLE
927 ) {
928 ++scope->iterator;
929 }
930
931 if (op == BSERIAL_OP_BLOB) {
932 BSERIAL_CHECK_STATUS(bserial_push_scope(ctx, BSERIAL_SCOPE_BLOB));
933 } else if (op == BSERIAL_OP_ARRAY) {
934 BSERIAL_CHECK_STATUS(bserial_push_scope(ctx, BSERIAL_SCOPE_ARRAY));
935 } else if (op == BSERIAL_OP_TABLE) {
936 BSERIAL_CHECK_STATUS(bserial_push_scope(ctx, BSERIAL_SCOPE_TABLE));
937 } else if (op == BSERIAL_OP_RECORD) {
938 BSERIAL_CHECK_STATUS(bserial_push_scope(ctx, BSERIAL_SCOPE_RECORD));
939 }
940
941 return BSERIAL_OK;
942}

944static inline bserial_status_t
945bserial_end_op(bserial_ctx_t* ctx, bserial_op_type_t op) {
946 BSERIAL_CHECK_STATUS(ctx->status);
947
948 // Record can have any nested bserial_end_op within its value but only the
949 // bserial_end_op that ends its own op should pop the scope.
950 if (
951 (ctx->scope->type == BSERIAL_SCOPE_BLOB && op == BSERIAL_OP_BLOB)
952 || (ctx->scope->type == BSERIAL_SCOPE_RECORD && op == BSERIAL_OP_RECORD)
953 ) {
954 BSERIAL_CHECK_STATUS(bserial_pop_scope(ctx));
955 }
956
957 // Auto pop when enough ops are executed.
958 // Array and table do not have an "end" function call and the number of
959 // elements are automatically tracked through bserial_end_op.
960 while (
961 (ctx->scope->type == BSERIAL_SCOPE_ARRAY
962 || ctx->scope->type == BSERIAL_SCOPE_TABLE)
963 && ctx->scope->iterator == ctx->scope->len
964 ) {
965 BSERIAL_CHECK_STATUS(bserial_pop_scope(ctx));
966 }
967
968 return BSERIAL_OK;
969}

1226
1227// small, fast 64 bit hash function (version 2).
1228//
1229// https://github.com/N-R-K/ChibiHash
1230//
1231// This is free and unencumbered software released into the public domain.
1232// For more information, please refer to <https://unlicense.org/>
1233
1234static inline uint64_t
1235bserial__chibihash64__load32le(const uint8_t *p) {
1236 return (uint64_t)p[0] << 0 | (uint64_t)p[1] << 8 |
1237 (uint64_t)p[2] << 16 | (uint64_t)p[3] << 24;
1238}
1239
1240static inline uint64_t
1241bserial__chibihash64__load64le(const uint8_t *p) {
1242 return bserial__chibihash64__load32le(p) | (bserial__chibihash64__load32le(p+4) << 32);
1243}
1244
1245static inline uint64_t
1246bserial__chibihash64__rotl(uint64_t x, int n) {
1247 return (x << n) | (x >> (-n & 63));
1248}
1249
1250static inline uint64_t
1251bserial__chibihash64(const void *keyIn, ptrdiff_t len, uint64_t seed) {
1252 const uint8_t *p = (const uint8_t *)keyIn;
1253 ptrdiff_t l = len;
1254
1255 const uint64_t K = UINT64_C(0x2B7E151628AED2A7); // digits of e
1256 uint64_t seed2 = bserial__chibihash64__rotl(seed-K, 15) + bserial__chibihash64__rotl(seed-K, 47);
1257 uint64_t h[4] = { seed, seed+K, seed2, seed2+(K*K^K) };
1258
1259 // depending on your system unrolling might (or might not) make things
1260 // a tad bit faster on large strings. on my system, it actually makes
1261 // things slower.
1262 // generally speaking, the cost of bigger code size is usually not
1263 // worth the trade-off since larger code-size will hinder inlinability
1264 // but depending on your needs, you may want to uncomment the pragma
1265 // below to unroll the loop.
1266 //#pragma GCC unroll 2
1267 for (; l >= 32; l -= 32) {
1268 for (int i = 0; i < 4; ++i, p += 8) {
1269 uint64_t stripe = bserial__chibihash64__load64le(p);
1270 h[i] = (stripe + h[i]) * K;
1271 h[(i+1)&3] += bserial__chibihash64__rotl(stripe, 27);
1272 }
1273 }
1274
1275 for (; l >= 8; l -= 8, p += 8) {
1276 h[0] ^= bserial__chibihash64__load32le(p+0); h[0] *= K;
1277 h[1] ^= bserial__chibihash64__load32le(p+4); h[1] *= K;
1278 }
1279
1280 if (l >= 4) {
1281 h[2] ^= bserial__chibihash64__load32le(p);
1282 h[3] ^= bserial__chibihash64__load32le(p + l - 4);
1283 } else if (l > 0) {
1284 h[2] ^= p[0];
1285 h[3] ^= p[l/2] | ((uint64_t)p[l-1] << 8);
1286 }
1287
1288 h[0] += bserial__chibihash64__rotl(h[2] * K, 31) ^ (h[2] >> 31);
1289 h[1] += bserial__chibihash64__rotl(h[3] * K, 31) ^ (h[3] >> 31);
1290 h[0] *= K; h[0] ^= h[0] >> 31;
1291 h[1] += h[0];
1292
1293 uint64_t x = (uint64_t)len * K;
1294 x ^= bserial__chibihash64__rotl(x, 29);
1295 x += seed;
1296 x ^= h[1];
1297
1298 x ^= bserial__chibihash64__rotl(x, 15) ^ bserial__chibihash64__rotl(x, 42);
1299 x *= K;
1300 x ^= bserial__chibihash64__rotl(x, 13) ^ bserial__chibihash64__rotl(x, 31);
1301
1302 return x;
1303}
1304
1305// }}}

1307static inline uint64_t
1308bserial_hash(const void* data, size_t len) {
1309 return bserial__chibihash64(data, (ptrdiff_t)len, 0);
1310}

1313static inline int32_t
1314bserial_lookup_index(uint64_t hash, int32_t exp, int32_t idx) {
1315 uint32_t mask = ((uint32_t)1 << exp) - 1;
1316 uint32_t step = (uint32_t)((hash >> (64 - exp)) | 1);
1317 return (idx + step) & mask;
1318}

1320static inline bserial_status_t
1321bserial_skip_next(bserial_ctx_t* ctx, uint32_t depth) {
1322 uint8_t marker;
1323 BSERIAL_CHECK_STATUS(bserial_peek_marker(ctx, &marker));
1324
1325 switch ((bserial_marker_t)marker) {
1326 case BSERIAL_UINT:
1327 {
1328 uint64_t u64;
1329 BSERIAL_CHECK_STATUS(bserial_uint(ctx, &u64));
1330 }
1331 break;
1332 case BSERIAL_SINT:
1333 {
1334 int64_t s64;
1335 BSERIAL_CHECK_STATUS(bserial_sint(ctx, &s64));
1336 }
1337 break;
1338 case BSERIAL_F32:
1339 {
1340 bserial_discard_marker(ctx);
1341 BSERIAL_CHECK_STATUS(ctx->status = bserial_skip(ctx->in, sizeof(float)));
1342 }
1343 break;
1344 case BSERIAL_F64:
1345 {
1346 bserial_discard_marker(ctx);
1347 BSERIAL_CHECK_STATUS(ctx->status = bserial_skip(ctx->in, sizeof(double)));
1348 }
1349 break;
1350 case BSERIAL_BLOB:
1351 {
1352 bserial_discard_marker(ctx);
1353 uint64_t len;
1354 BSERIAL_CHECK_STATUS(bserial_read_uint(&len, ctx->in));
1355 BSERIAL_CHECK_STATUS(ctx->status = bserial_skip(ctx->in, len));
1356 }
1357 break;
1358 case BSERIAL_SYM_DEF:
1359 case BSERIAL_SYM_REF:
1360 {
1361 const char* sym;
1362 uint64_t sym_len;
1363 BSERIAL_CHECK_STATUS(bserial_symbol(ctx, &sym, &sym_len));
1364 }
1365 break;
1366 case BSERIAL_ARRAY:
1367 {
1368 bserial_discard_marker(ctx);
1369
1370 uint64_t len;
1371 BSERIAL_CHECK_STATUS(bserial_read_uint(&len, ctx->in));
1372 if (len > 0 && depth == 0) { return bserial_malformed(ctx); }
1373
1374 for (uint64_t i = 0; i < len; ++i) {
1375 BSERIAL_CHECK_STATUS(bserial_skip_next(ctx, depth - 1));
1376 }
1377 }
1378 break;
1379 case BSERIAL_TABLE:
1380 {
1381 bserial_discard_marker(ctx);
1382
1383 uint64_t num_rows;
1384 BSERIAL_CHECK_STATUS(bserial_read_uint(&num_rows, ctx->in));
1385 if (num_rows > 0 && depth == 0) { return bserial_malformed(ctx); }
1386
1387 if (num_rows > 0) {
1388 uint64_t num_cols;
1389 BSERIAL_CHECK_STATUS(bserial_read_uint(&num_cols, ctx->in));
1390
1391 for (uint64_t i = 0; i < num_cols; ++i) {
1392 const char* sym;
1393 uint64_t sym_len;
1394 BSERIAL_CHECK_STATUS(bserial_symbol(ctx, &sym, &sym_len));
1395 }
1396
1397 for (uint64_t i = 0; i < num_rows; ++i) {
1398 for (uint64_t j = 0; j < num_cols; ++j) {
1399 BSERIAL_CHECK_STATUS(bserial_skip_next(ctx, depth - 1));
1400 }
1401 }
1402 }
1403 }
1404 break;
1405 case BSERIAL_RECORD:
1406 {
1407 bserial_discard_marker(ctx);
1408
1409 uint64_t num_cols;
1410 BSERIAL_CHECK_STATUS(bserial_read_uint(&num_cols, ctx->in));
1411
1412 for (uint64_t i = 0; i < num_cols; ++i) {
1413 const char* sym;
1414 uint64_t sym_len;
1415 BSERIAL_CHECK_STATUS(bserial_symbol(ctx, &sym, &sym_len));
1416 }
1417
1418 for (uint64_t i = 0; i < num_cols; ++i) {
1419 BSERIAL_CHECK_STATUS(bserial_skip_next(ctx, depth - 1));
1420 }
1421 }
1422 break;
1423 default:
1424 return bserial_malformed(ctx);
1425 }
1426
1427 return BSERIAL_OK;
1428}

1430static inline bool
1431bserial_probe_next_record_field(bserial_ctx_t* ctx) {
1432 bserial_scope_t* scope = ctx->scope;
1433
1434 while (scope->iterator < scope->len) {
1435 if (scope->record_schema[scope->iterator].field_name != NULL) {
1436 return true;
1437 } else {

1439 bserial_skip_next(

1441 ctx->config.max_depth - (uint32_t)(ctx->scope - ctx->scope_first)

1443 ) {
1444 return false;
1445 }
1446 ++scope->iterator;
1447 }
1448 }

1450 bserial_end_op(ctx, BSERIAL_OP_RECORD);
1451 return false;

1454bserial_status_t
1455bserial_symbol(bserial_ctx_t* ctx, const char** buf, uint64_t* len) {
1456 BSERIAL_CHECK_STATUS(bserial_begin_op(ctx, BSERIAL_OP_SYMBOL));
1457
1458 if (bserial_mode(ctx) == BSERIAL_MODE_READ) {
1459 uint8_t marker;
1460 BSERIAL_CHECK_STATUS(bserial_read_marker(ctx, &marker));
1461
1462 if (marker == BSERIAL_SYM_DEF) {
1463 if (ctx->num_symbols >= ctx->config.max_num_symbols) { return bserial_malformed(ctx); }
1464
1465 uint64_t symbol_len;
1466 BSERIAL_CHECK_STATUS(ctx->status = bserial_read_uint(&symbol_len, ctx->in));
1467 if (symbol_len > ctx->config.max_symbol_len) { return bserial_malformed(ctx); }
1468
1469 BSERIAL_CHECK_STATUS(ctx->status = bserial_read(ctx->in, ctx->strpool, symbol_len));
1470 ctx->strpool[symbol_len] = '\0';
1471
1472 ctx->symtab[ctx->num_symbols] = (bserial_symbol_t){
1473 .buf = ctx->strpool,
1474 .len = symbol_len,
1475 };
1476
1477 *buf = ctx->strpool;
1478 *len = symbol_len;
1479
1480 ctx->num_symbols += 1;
1481 ctx->strpool += (symbol_len + 1);
1482 } else if (marker == BSERIAL_SYM_REF) {
1483 uint64_t symbol_id;
1484 BSERIAL_CHECK_STATUS(ctx->status = bserial_read_uint(&symbol_id, ctx->in));
1485 if (symbol_id >= ctx->num_symbols) { return bserial_malformed(ctx); }
1486
1487 *buf = ctx->symtab[symbol_id].buf;
1488 *len = ctx->symtab[symbol_id].len;
1489 } else {
1490 return bserial_malformed(ctx);
1491 }
1492 } else {
1493 uint64_t symbol_len = *len;
1494 if (symbol_len > ctx->config.max_symbol_len) { return bserial_malformed(ctx); }
1495
1496 uint64_t symbol_hash = bserial_hash(*buf, symbol_len);
1497 for (int32_t i = (int32_t)symbol_hash;;) {
1498 i = bserial_lookup_index(symbol_hash, ctx->symtab_exp, i);
1499 int32_t index = ctx->symtab_index[i];
1500 if (index == 0) {
1501 if (ctx->num_symbols >= ctx->config.max_num_symbols) {
1502 return bserial_malformed(ctx);
1503 }
1504 memcpy(ctx->strpool, *buf, symbol_len);
1505 ctx->strpool[symbol_len] = '\0';
1506
1507 ctx->symtab[ctx->num_symbols] = (bserial_symbol_t){
1508 .buf = ctx->strpool,
1509 .len = symbol_len,
1510 };
1511 ctx->symtab_index[i] = ctx->num_symbols + 1;
1512 ctx->num_symbols += 1;
1513 ctx->strpool += (symbol_len + 1);
1514
1515 uint8_t marker = BSERIAL_SYM_DEF;
1516 BSERIAL_CHECK_STATUS(ctx->status = bserial_write(ctx->out, &marker, sizeof(marker)));
1517 BSERIAL_CHECK_STATUS(ctx->status = bserial_write_uint(symbol_len, ctx->out));
1518 BSERIAL_CHECK_STATUS(ctx->status = bserial_write(ctx->out, *buf, symbol_len));
1519
1520 break;
1521 } else if (
1522 ctx->symtab[index - 1].len == symbol_len
1523 && memcmp(ctx->symtab[index - 1].buf, *buf, symbol_len) == 0
1524 ) {
1525 uint8_t marker = BSERIAL_SYM_REF;
1526 BSERIAL_CHECK_STATUS(ctx->status = bserial_write(ctx->out, &marker, sizeof(marker)));
1527 BSERIAL_CHECK_STATUS(ctx->status = bserial_write_uint((uint64_t)index - 1, ctx->out));
1528
1529 break;
1530 }
1531 }
1532 }
1533
1534 return bserial_end_op(ctx, BSERIAL_OP_SYMBOL);
1535}

1563bool
1564bserial_record(bserial_ctx_t* ctx, void* record) {
1565 if (ctx->status != BSERIAL_OK) { return false; }
1566
1567 bserial_scope_t* scope = ctx->scope;
1568
1569 if (bserial_mode(ctx) == BSERIAL_MODE_READ) {
1570 if (scope->type == BSERIAL_SCOPE_RECORD && record == scope->record_addr) {
1571 switch (scope->record_mode) {
1572 case BSERIAL_RECORD_KEY_IO:
1573 scope->record_mode = BSERIAL_RECORD_VALUE_IO;
1574 scope->iterator = 0;
1575 return bserial_probe_next_record_field(ctx);
1576 case BSERIAL_RECORD_VALUE_IO:
1577 if (scope->iterator == scope->len) {
1578 bserial_end_op(ctx, BSERIAL_OP_RECORD);
1579 return false;
1580 } else {
1581 return bserial_probe_next_record_field(ctx);
1582 }
1583 default:
1584 bserial_malformed(ctx);
1585 return false;
1586 }
1587 } else {
1588 bserial_scope_t* parent_scope = scope;
1589 if (bserial_begin_op(ctx, BSERIAL_OP_RECORD) != BSERIAL_OK) {
1590 return false;
1591 }
1592 scope = ctx->scope;
1593 scope->record_addr = record;
1594
1595 if (parent_scope->type != BSERIAL_SCOPE_TABLE) {
1596 uint8_t marker;
1597 if ((bserial_read_marker(ctx, &marker)) != BSERIAL_OK) {
1598 return false;
1599 }
1600 if (marker != BSERIAL_RECORD) {
1601 bserial_malformed(ctx);
1602 return false;
1603 }
1604 }
1605
1606 // In case of a table, allocate schema at the table's scope.
1607 bserial_scope_t* schema_scope =
1608 parent_scope->type == BSERIAL_SCOPE_TABLE
1609 ? parent_scope
1610 : scope;
1611
1612 // Schema discovery only happens for the first row of a table
1613 if (parent_scope->type != BSERIAL_SCOPE_TABLE || parent_scope->iterator == 1) {
1614 scope->record_mode = BSERIAL_RECORD_KEY_IO;
1615
1616 uint64_t num_fields;
1617 if ((ctx->status = bserial_read_uint(&num_fields, ctx->in)) != BSERIAL_OK) {
1618 return false;
1619 }
1620 if (num_fields > ctx->config.max_record_fields) {
1621 bserial_malformed(ctx);
1622 return false;
1623 }
1624
1625 for (uint64_t i = 0; i < num_fields; ++i) {
1626 schema_scope->record_schema[i].field_name = NULL;
1627 }
1628
1629 schema_scope->record_width = num_fields;
1630 for (uint64_t i = 0; i < num_fields; ++i) {
1631 const char* symbol;
1632 uint64_t symbol_len;
1633 scope->iterator = i;
1634 if (bserial_symbol(ctx, &symbol, &symbol_len) != BSERIAL_OK) {
1635 return false;
1636 }
1637
1638 schema_scope->record_schema[i].symbol = symbol;
1639 schema_scope->record_schema[i].symbol_len = symbol_len;
1640 }
1641 scope->iterator = 0;
1642 } else {
1643 scope->record_mode = BSERIAL_RECORD_VALUE_IO;
1644 }
1645
1646 scope->record_schema = schema_scope->record_schema;
1647 scope->len = schema_scope->record_width;
1648
1649 if (scope->record_mode == BSERIAL_RECORD_VALUE_IO) {
1650 return bserial_probe_next_record_field(ctx);
1651 } else {
1652 return true;
1653 }
1654 }
1655 } else {
1656 if (scope->type == BSERIAL_SCOPE_RECORD && scope->record_addr == record) {
1657 switch (scope->record_mode) {
1658 case BSERIAL_RECORD_MEASURE_WIDTH:
1659 scope->record_mode = BSERIAL_RECORD_KEY_IO;
1660 ctx->status = bserial_write_uint(scope->len, ctx->out);
1661 return ctx->status == BSERIAL_OK;
1662 case BSERIAL_RECORD_KEY_IO:
1663 scope->record_mode = BSERIAL_RECORD_VALUE_IO;
1664 scope->iterator = 0;
1665 return true;
1666 case BSERIAL_RECORD_VALUE_IO:
1667 bserial_end_op(ctx, BSERIAL_OP_RECORD);
1668 return false;
1669 default:
1670 bserial_malformed(ctx);
1671 return false;
1672 }
1673 } else {
1674 bserial_scope_t* parent_scope = scope;
1675 if (bserial_begin_op(ctx, BSERIAL_OP_RECORD) != BSERIAL_OK) {
1676 return false;
1677 }
1678 scope = ctx->scope;
1679 scope->record_addr = record;
1680
1681 if (parent_scope->type != BSERIAL_SCOPE_TABLE || parent_scope->iterator == 1) {
1682 scope->record_mode = BSERIAL_RECORD_MEASURE_WIDTH;
1683
1684 if (parent_scope->type != BSERIAL_SCOPE_TABLE) {
1685 uint8_t marker = BSERIAL_RECORD;
1686 if ((ctx->status = bserial_write(ctx->out, &marker, sizeof(marker))) != BSERIAL_OK) {
1687 return false;
1688 }
1689 }
1690 } else {
1691 scope->record_mode = BSERIAL_RECORD_VALUE_IO;
1692 }
1693
1694 return true;
1695 }
1696 }

1700bserial_key(bserial_ctx_t* ctx, const char* name, uint64_t len) {
1701 if (ctx->status != BSERIAL_OK) { return false; }
1702
1703 bserial_scope_t* scope = ctx->scope;
1704
1705 if (scope->type != BSERIAL_SCOPE_RECORD) {
1706 bserial_malformed(ctx);
1707 return false;
1708 }
1709
1710 if (bserial_mode(ctx) == BSERIAL_MODE_READ) {
1711 switch (scope->record_mode) {
1712 case BSERIAL_RECORD_KEY_IO:
1713 for (uint64_t i = 0; i < scope->len; ++i) {
1714 if (
1715 scope->record_schema[i].symbol_len == len
1716 && strncmp(scope->record_schema[i].symbol, name, len) == 0
1717 ) {
1718 scope->record_schema[i].field_name = name;
1719 }
1720 }
1721 ++scope->iterator;
1722 return false;
1723 case BSERIAL_RECORD_VALUE_IO:
1724 if (name == scope->record_schema[scope->iterator].field_name) {
1725 ++scope->iterator;
1726 return true;
1727 } else {
1728 return false;
1729 }
1730 default:
1731 bserial_malformed(ctx);
1732 return false;
1733 }
1734 } else {
1735 switch (scope->record_mode) {
1736 case BSERIAL_RECORD_MEASURE_WIDTH:
1737 ++scope->len;
1738 return false;
1739 case BSERIAL_RECORD_KEY_IO:
1740 ++scope->iterator;
1741 bserial_symbol(ctx, &name, &len);
1742 return false;
1743 case BSERIAL_RECORD_VALUE_IO:
1744 ++scope->iterator;
1745 return true;
1746 default:
1747 bserial_malformed(ctx);
1748 return false;
1749 }
1750 }
